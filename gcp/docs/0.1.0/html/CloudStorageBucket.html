    // CloudStorageBucket
schema CloudStorageBucket {
    string  name                             // Globally unique bucket name
    string  location = "US"                  // The location of the bucket (e.g., US, EU, us-central1)
    @allowed(["STANDARD", "NEARLINE", "COLDLINE", "ARCHIVE"])
    string  storageClass = "STANDARD"        // The storage class for the bucket
    boolean versioning = false               // Enable object versioning
    object  labels                           // Labels to apply to the bucket

    @cloud(importable)

    string  timeCreated                      // The creation time of the bucket
}
