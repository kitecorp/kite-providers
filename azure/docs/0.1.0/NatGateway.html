NatGateway

Documentation for NatGateway resource.

Import

import NatGateway from "azure/networking/NatGateway.kite"

Example

resource NatGateway example {
    name                 = "example-value"
    resourceGroup        = "example-value"
    location             = "example-value"
    idleTimeoutInMinutes = 4
    publicIpAddressIds   = ["item1", "item2"]
    publicIpPrefixIds    = ["item1", "item2"]
    skuName              = "Standard"
    tags                 = { key = "value" }
}
