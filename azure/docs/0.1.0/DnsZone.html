DnsZone

Documentation for DnsZone resource.

Import

import DnsZone from "azure/dns/DnsZone.kite"

Example

resource DnsZone example {
    name                          = "example-value"
    resourceGroup                 = "example-value"
    zoneType                      = "Public"
    registrationVirtualNetworkIds = ["item1", "item2"]
    resolutionVirtualNetworkIds   = ["item1", "item2"]
    tags                          = { key = "value" }
}
