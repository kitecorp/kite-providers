ManagedDisk

Documentation for ManagedDisk resource.

Import

import ManagedDisk from "azure/storage/ManagedDisk.kite"

Example

resource ManagedDisk example {
    name                = "example-value"
    resourceGroup       = "example-value"
    location            = "example-value"
    size                = 42
    sku                 = "StandardSSD_LRS"
    tier                = "example-value"
    diskIops            = 42
    diskMbps            = 42
    osType              = "Linux"
    sourceResourceId    = "example-value"
    sourceUri           = "example-value"
    storageAccountId    = "example-value"
    zone                = "example-value"
    burstingEnabled     = false
    networkAccessPolicy = "AllowAll"
    diskAccessId        = "example-value"
    tags                = { key = "value" }
}
