BlobContainer

Documentation for BlobContainer resource.

Import

import BlobContainer from "azure/storage/BlobContainer.kite"

Example

resource BlobContainer example {
    name                           = "example-value"
    storageAccountName             = "example-value"
    resourceGroup                  = "example-value"
    publicAccess                   = "None"
    metadata                       = { key = "value" }
    immutableStorageWithVersioning = true
    defaultEncryptionScope         = "example-value"
    denyEncryptionScopeOverride    = true
}
