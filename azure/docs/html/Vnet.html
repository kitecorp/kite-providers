    // Vnet
schema Vnet {
    string name                          // The name of the virtual network
    string resourceGroup                 // The Azure resource group name
    string location                      // The Azure region/location (e.g., eastus, westeurope)
    list addressSpaces                 // The address spaces for the VNet in CIDR notation
    list dnsServers                    // Custom DNS servers for the VNet. If not specified, Azure-provided DNS is used
    boolean enableDdosProtection = false  // Enable DDoS protection for the VNet
    boolean enableVmProtection = false    // Enable VM protection for the VNet
    map tags                          // Tags to apply to the VNet
